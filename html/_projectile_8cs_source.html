161 public void UpdateProjectile(double deltaTime)
162 {
163 _time += deltaTime;
164 double rad = _angle * Math.PI / 180.0;
165 int newX = _position.X + (int)(_initialVelocity * Math.Cos(rad) * _time);
166 int newY = _position.Y - (int)((_initialVelocity * Math.Sin(rad) * _time) - (0.5 * _gravity * _time * _time));
167
168 // Affichage de debogage temporaire
169 Console.SetCursorPosition(80, 30);
170 Console.Write($"X: {newX} Y: {newY} ");
171
172 _position = new Position(newX, newY);
173
174 //Debug
175 int windowWidth;
176 int windowHeight;
177 if (Environment.UserInteractive)
178 {
179 windowWidth = Console.WindowWidth;
180 windowHeight = Console.WindowHeight;
181 }
182 else
183 {
184 // En environnement non interactif (tests unitaires), utiliser des valeurs par défaut
185 windowWidth = Config.SCREEN_WIDTH;
186 windowHeight = Config.SCREEN_HEIGHT;
187 }
188 //Fin debug
189
190 // Si le projectile sort de la fenêtre ou "touche le sol", on le désactive.
191 if (newY >= windowHeight || newX >= windowWidth || newX < 0)
192 {
193 _isActive = false;
194 }
195 }
196
201 public void DisplayProjectile()
202 {
203 if (_isActive)
204 {
205 // Efface le point précédent, si existant
206 if (_prevX != -1 && _prevY != -1)
207 {
208 Console.SetCursorPosition(_prevX, _prevY);
209 Console.Write(" ");
210 }
211 try
212 {
213 // Affiche le projectile à sa nouvelle position
214 Console.SetCursorPosition(_position.X, _position.Y);
215 Console.Write(_projectileChar);
216 // Mémorise la nouvelle position
217 _prevX = _position.X;
218 _prevY = _position.Y;
219 }
220 catch (ArgumentOutOfRangeException)
221 {
222
223 }
224 }
225
226 // Si le projectile n'est pas actif, efface la dernière position
227 else if (_prevX != -1 && _prevY != -1)
228 {
229 Console.SetCursorPosition(_prevX, _prevY);
230 Console.Write(" ");
231 }
232 }
233 }
