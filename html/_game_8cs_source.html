102 public void RunGame()
103 {
104 // Configure la console pour le jeu
105 Config.ConfigJeu();
106
107 //Créer les tours
108 _tour1.CréerTour();
109 _tour2.CréerTour();
110
111 while (true)
112 {
113 // Afficher les éléments du jeu
114 _joueur1.AffichageJoueur();
115 _tour1.AffichageTour();
116 _joueur2.AffichageJoueur();
117 _tour2.AffichageTour();
118 // Affiche les scores à chaque début de tour
119 _scoreJ1.DisplayScore();
120 _scoreJ2.DisplayScore();
121
122 // Détermine quel joueur joue et gère son tir
123 if (_isPlayer1Turn)
124 {
125 int angle = _interfaceTirJ1.SelectAngle();
126 int power = _interfaceTirJ1.SelectPower();
127 Projectile projectile = new Projectile(angle, power, _interfaceTirJ1.Position, true, 0);
128 _projectiles.Add(projectile);
129 }
130 else
131 {
132 int angle = _interfaceTirJ2.SelectAngle();
133 int power = _interfaceTirJ2.SelectPower();
134 Projectile projectile = new Projectile(angle, power, _interfaceTirJ2.Position, true, 0);
135 _projectiles.Add(projectile);
136 }
137
138 // Anime le projectile jusqu'à sa fin de vie
139 while (_projectiles.Count > 0)
140 {
141 Console.SetCursorPosition(80, 20);
142 Console.Write(_collisionDetected);
143
144 // Parcourir une copie de la liste pour éviter de modifier la liste pendant l'itération
145 foreach (var proj in new List<Projectile>(_projectiles))
146 {
147 if (proj.IsActive)
148 {
149 proj.UpdateProjectile(0.05); // Simule 0,05 seconde d'écoulement
150 proj.DisplayProjectile();
151
152 // Vérifier la collision avec le joueur adverse selon le tour actuel
153 _collisionDetected = false;
154 if (_isPlayer1Turn)
155 {
156 // Si c'est le tour du joueur 1, le projectile vise le joueur 2
157 if (_joueur2.HitBox.isTouched(proj.Position.X, proj.Position.Y))
158 {
159 _collisionDetected = true;
160 _joueur2.TakeDamage();
161 proj.IsActive = false;
162 }
163 }
164 else
165 {
166 // Si c'est le tour du joueur 2, le projectile vise le joueur 1
167 if (_joueur1.HitBox.isTouched(proj.Position.X, proj.Position.Y))
168 {
169 _collisionDetected = true;
170 _joueur1.TakeDamage();
171 proj.IsActive = false;
172 }
173 }
174
175 //Détection si les tours sont touchées
176 if (_tour1.Hitbox.isTouched(proj.Position.X, proj.Position.Y))
177 {
178 _tour1.TourCollision();
179 }
180 if(_tour2.Hitbox.isTouched(proj.Position.X,proj.Position.Y))
181 {
182 _tour2.TourCollision();
183 }
184
185 // Si le projectile n'est plus actif (collision ou hors écran) on le supprime de la liste
186 if (!proj.IsActive)
187 {
188 _projectiles.Remove(proj);
189 }
190 }
191 }
192 Thread.Sleep(100); // Attend pour permettre l'animation
193 }
194 // Changement de joueur
195 _isPlayer1Turn = !_isPlayer1Turn;
196
197 }
198 }
199 }
